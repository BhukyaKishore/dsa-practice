class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
        int totalproduct=1;
        vector<int> answer;
        for(int i=0;i<nums.size();i++){
            if(nums[i]!=0){
                totalproduct*=nums[i];
            }
        }
        for(int i=0;i<nums.size();i++){
            if(nums[i]!=0){
                answer[i]=totalproduct/nums[i];
            }else{
                answer[i]=totalproduct;
            }
        }
        return answer;
    }
};
